/* 시퀀스 종류
      BOARD_SEQ : 모든 공지사항 ( NOTICE, REVIEW, QNA, UPLOAD )
      REPORT_SEQ : 신고하기 게시판 ( REPORT, BLACKLIST )
      MEMO_SEQ : 청첩장 방명록 ( MEMO )
*/

CREATE TABLE common
(
    ID       VARCHAR2(400)    NOT NULL, 
    PW       VARCHAR2(400)    NULL, 
    NAME     VARCHAR2(400)    NULL, 
    ADDR     VARCHAR2(1000)    NULL, 
    PHONE    VARCHAR2(1000)    NULL, 
    MAIL     VARCHAR2(1000)    NULL, 
    JOB      VARCHAR2(20)    NULL, 
    CONSTRAINT COMMON_PK PRIMARY KEY (ID)
)

CREATE TABLE member
(
    ID             VARCHAR2(400)    NOT NULL, 
    BIRTH          VARCHAR2(1000)    NULL, 
    GENDER         VARCHAR2(20)    NULL, 
    WEDDING_DAY    VARCHAR2(1000)    NULL, 
    CONSTRAINT MEMBER_PK PRIMARY KEY (ID)
)

CREATE TABLE COMPANY
(
    ID                VARCHAR2(400)    NOT NULL, 
    COMPANY_NAME      VARCHAR2(400)    NULL, 
    COMPANY_TEL       VARCHAR2(1000)    NULL, 
    COMPANY_NUMBER    VARCHAR2(400)    NULL, 
    CONSTRAINT COMPANY_PK PRIMARY KEY (ID)
)

CREATE TABLE extra
(
    ID           VARCHAR2(400)    NOT NULL, 
    REGION       VARCHAR2(400)    NULL, 
    TYPE         VARCHAR2(400)    NULL, 
    MEAL_COST    NUMBER(8)    NULL, 
    MEAL_MENU    VARCHAR2(400)    NULL, 
    VISITOR      NUMBER(8)    NULL, 
    SUBWAY       VARCHAR2(400)    NULL, 
    HALL_COST    NUMBER(8)    NULL, 
    HALL_NAME    VARCHAR2(400)    NULL, 
    CONSTRAINT EXTRA_PK PRIMARY KEY (ID)
)


CREATE TABLE RESERVATION
(
    ID            VARCHAR2(400)    NOT NULL, 
    PRE_DATE      VARCHAR2(1000)    NULL, 
    PRE_REGION    VARCHAR2(400)    NULL, 
    CONTENTS      VARCHAR2(4000)    NULL, 
    CONSTRAINT RESERVATION_PK PRIMARY KEY (ID)
)


CREATE TABLE BOOKMARK
(
    ID                VARCHAR2(400)    NOT NULL, 
    INVITATION_NUM    NUMBER(8)             NULL, 
    WEDDING_NUM       NUMBER(8)             NULL, 
    CONSTRAINT BOOKMARK_PK PRIMARY KEY (ID)
)


CREATE TABLE BALCKLIST
(
    MEMBER_ID       VARCHAR2(400)    NOT NULL, 
    COMPANY_NAME    VARCHAR2(400)    NULL, 
    COMPANY_ID      VARCHAR2(400)    NULL, 
    NUM             NUMBER(8)        NULL   
)


CREATE TABLE REPORT
(
    NUM             NUMBER(8)       NOT NULL, 
    PW              VARCHAR2(400)    NULL, 
    ID              VARCHAR2(400)    NULL, 
    TITLE           VARCHAR2(1000)    NULL, 
    CONTENTS        VARCHAR2(4000)    NULL, 
    COMPANY_NAME    VARCHAR2(400)    NULL   
)

CREATE TABLE INVITATION
(
    INVITATION_NUM    NUMBER(8)    NOT NULL, 
    ID                VARCHAR2(400)     NOT NULL, 
    IMAGE_NUM         VARCHAR2(400)     NULL, 
    M_NAME            VARCHAR2(400)     NULL, 
    W_NAME            VARCHAR2(400)     NULL, 
    M_TEL             VARCHAR2(800)     NULL, 
    W_TEL             VARCHAR2(800)     NULL, 
    WEDDING_DATE      VARCHAR2(1000)     NULL, 
    WEDDING_ADDR      VARCHAR2(1000)     NULL, 
    GREETING          VARCHAR2(4000)     NULL, 
    CONSTRAINT INVITATION_PK PRIMARY KEY (INVITATION_NUM)
)


CREATE TABLE MEMO
(
    NUM         NUMBER(8)             NOT NULL, 
    ID          VARCHAR2(400)    NOT NULL, 
    NAME        VARCHAR2(400)    NULL, 
    PASSWORD    VARCHAR2(400)    NULL, 
    CONTENTS    VARCHAR2(4000)    NULL, 
    CONSTRAINT MEMO_PK PRIMARY KEY (NUM)
)


CREATE TABLE notice
(
    NUM         NUMBER(8)         NOT NULL, 
    TITLE       VARCHAR2(1000)    NULL, 
    WRITER      VARCHAR2(400)    NULL, 
    CONTENTS    VARCHAR2(4000)    NULL, 
    REG_DATE    VARCHAR2(1000)    NULL, 
    HIT         NUMBER(8)             NULL, 
    CONSTRAINT BOARD_PK PRIMARY KEY (NUM)
)


CREATE TABLE REVIEW
(
    NUM          NUMBER(8)       NOT NULL, 
    STARSCORE    NUMBER(8)       NULL, 
    KIND         VARCHAR2(400)    NULL   
)


CREATE TABLE QNA
(
    NUM          NUMBER(4)       NOT NULL, 
    REF          NUMBER(4)       NULL, 
    STEP         NUMBER(4)       NULL, 
    DEPTH        NUMBER(4)       NULL, 
    PASSWORD     VARCHAR2(400)    NULL, 
    OPENCHECK    VARCHAR2(400)    NULL   
)

CREATE TABLE UPLOAD
(
    NUM      NUMBER(8)         NOT NULL, 
    FNAME    VARCHAR2(2000)    NULL, 
    ONAME    VARCHAR2(2000)    NULL   
)

CREATE TABLE REPLE
(
    NUM      NUMBER(8)    NOT NULL, 
    REF      NUMBER(8)    NULL, 
    STEP     NUMBER(8)    NULL, 
    DEPTH    NUMBER(8)    NULL   
)


CREATE SEQUENCE BOARD_SEQ INCREMENT BY 1 START WITH 1;
CREATE SEQUENCE REPORT_SEQ INCREMENT BY 1 START WITH 1;
CREATE SEQUENCE MEMO_SEQ INCREMENT BY 1 START WITH 1;
